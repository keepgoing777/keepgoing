    배열 2배값 출력 const doublemap = arr.map((value, index, array)=>{return value*2;}); console.log(doublemap);
    배열 짝수 출력 const evenfilter=arr.filter((value, index, array)=>{return value%2===0;}); console.log(evenfilter);
    배열 총합 출력 const sumReduce = arr.reduce((acc, cur, idx, arr)=> {return acc+cur;},0); console.log(sumReduce);
